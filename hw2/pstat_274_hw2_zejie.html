# Load the necessary library
library(stats)

# Set seed for reproducibility
set.seed(123)

# Simulate the MA(2) process for 300 values
n <- 300
Z <- rnorm(n)
X <- stats::filter(Z, filter=c(1,2,-8), method="convolution")
X <- X[!is.na(X)]  # Remove NA values

# Compute and plot the sample ACF for 300 values
acf_300 <- acf(X, plot=TRUE, main="Sample ACF with n=300")

# Extract autocorrelation value at lag 2 for both simulations
rho_2_300 <- acf_300$acf[3]
rho_2_300
## [1] -0.1657309
